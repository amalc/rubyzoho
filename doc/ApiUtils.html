def self.camelize_with_space(str)
  str.split('_').map {|w| w.capitalize}.join(' ')
end

def self.symbol_to_string(sym)
  sym.class == Symbol ? self.camelize_with_space(sym.to_s) : self.camelize_with_space(sym)
end
